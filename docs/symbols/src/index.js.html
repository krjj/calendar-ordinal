 46 /**
 47  * Checks if the gregorian year has leap day
 48  * @param {number} year - Year to check
 49  * @returns {boolean}
 50  * @example
 51  * isLeapYearGregorian(2017)
 52  */
 53 function isLeapYearGregorian(year) {
 54     return ((year % 4 == 0) && (year % 100 != 0)) || (year % 400 == 0)

 89 /**
 90  * Get date from Ordinal Date
 91  * @param {number} day Day number
 92  * @param {boolean} [isLeapYear=false] True if leap year
 93  * @param {number} [adjust=0] Substract no of days from the year
 94  * @returns {Object} 
 95  * @example
 96  * getDateFromOrdinal(244,false,0)
 97  */
 98 function getDateFromOrdinal(od, isLeapYear = false, adjust = 0) {
 99 
100     od -= adjust
101     monthCount[2] = isLeapYear === true ? 29 : 28
102 
103     let m, d, i = 0
104 
105     for (e of monthCount) { // iterate through the months
106         if (e != 0) {
107             od = od - e
108             if (od > 0) {
109                 // Ordinal date not in this month
110             } else {
111                 m = i
112                 d = (e) + (od)
113                 break
114             }
115         }
116         i++
117     }
118 
119     return { month: m, monthLong: month[m], day: d, dayWithSuffix: d + getSuffixDayOfMonth(d) }
120 }

123 function init(year = 2017, od = 244, mode = 'cal') {
124 
125     if (argv.y === undefined) { console.warn('\x1b[33m', 'Fallback to default year. Argument not passed -y', year, '\x1b[0m') }

128             console.error('\x1b[31m', 'Year must be an integer', '\x1b[0m')

131             year = argv.y
132         }
133     }
134 
135 
136     if (argv.o === undefined) { console.warn('\x1b[33m', 'Fallback to default ordinal date. Argument not passed -o', od, '\x1b[0m') }
137     else {
138         if (!Number.isInteger(argv.o)) {
139             console.error('\x1b[31m', 'Ordinal Date must be an integer', '\x1b[0m')
140             process.exit()
141         } else {
142             od = argv.o
143         }
144     }
145 
146     if (argv.m === undefined) { console.warn('\x1b[33m', 'Fallback to default mode.  Argument not passed -m', mode, '\x1b[0m') }
147     else {
148         if (argv.m === 'cal') {
149             mode = argv.m
150         } else if (argv.m === 'gcal') {
151             mode = argv.m
152         } else {
153             console.error('\x1b[31m', 'Invalid mode', argv.m, '\x1b[0m')
154             process.exit()
155         }
156     }
157 
158     ordinalDay = od
159 
160     console.info('\x1b[32m', 'Gregorian Reformation Period.', gregorianReform, '\x1b[0m')
161     console.info('\x1b[32m', 'Ordinal Date.', year + "-" + od, 'ISO 8601 YYYY-DDD format.', '\x1b[0m')
162 
163     let isLeap = isJulianYear(year) === true ? isLeapYearJulian(year) : isLeapYearGregorian(year)
164     console.info('\x1b[32m', year, isLeap ? 'is leap year' : 'is not leap year', isJulianYear(year) ? 'is julian year' : 'is gregorian year', '\x1b[0m')
165     let adj = 0;
166 
167     
168     if (year === gregorianReform[0] && mode == 'cal') {
169         console.info('\x1b[34m', year, 'Overide. This is year of gregorian reform.', '\x1b[0m')
170         adj = gregorianReform[3]
171     }
172 
173     let result = getDateFromOrdinal(ordinalDay, isLeap, adj)
174     return result.dayWithSuffix + ' ' + result.monthLong + ' ' + year
